CREATE FUNCTION [workcube_devcatalyst_retail].[fnc_get_gunluk_satis_stok]
(
@sno int,
@dept_id int,
@date1 datetime,
@yil nvarchar(4)
)
RETURNS float
AS
BEGIN

	DECLARE @SQL NVARCHAR(max), @ParmDefinition NVARCHAR(1024)
	DECLARE @ortalama float = 0
	SET @SQL =       N'	SELECT
        @ortalama_c = ISNULL(SUM(STOCK_OUT),0)
        FROM
			workcube_devcatalyst_' + @yil + '_1.STOCKS_ROW
        WHERE
            STOCK_ID = @stock_id AND 
            PROCESS_DATE= @process_date AND 
            PROCESS_TYPE IN (67,52,70) AND
            STORE = @store'
            
	SET @ParmDefinition = N'@stock_id int,
						@process_date datetime,
                        @store int,
                        @ortalama_c float OUTPUT
                        '
	EXECUTE sp_executeSQL      -- Dynamic T-SQL
            @SQL,
            @ParmDefinition,
            @stock_id = @sno,
            @store = @dept_id,
            @process_date = @date1,
            @ortalama_c = @ortalama OUTPUT
return(@ortalama)
END

CREATE FUNCTION [workcube_devcatalyst_retail].[fnc_get_ortalama_satis_stok_all]
(
@sno int,
@date1 datetime,
@date2 datetime
)
RETURNS float 
AS
BEGIN
  DECLARE @Result float
  SET @Result = 	(
			SELECT
				ROUND(SUM(GUNLUK_SATIS) / COUNT(TARIH),2) AS ORT
			FROM
				(
					SELECT
						PPD.TARIH,
						ISNULL(DAILY_STOCK,0) AS GUNLUK_STOK,
						ISNULL(DAILY_SALES,0) GUNLUK_SATIS
					FROM
						PRODUCT_PRICE_DATES PPD,
						PRODUCT_ACTIONS_DAILY_NOTE PADN
					WHERE
						PADN.STOCK_ID = @sno AND
						PADN.YIL = PPD.YIL AND 
						PADN.AY = PPD.AY AND 
						PADN.GUN = PPD.GUN AND
						PPD.TARIH BETWEEN @date1 AND @date2
				
			) A
		WHERE
			GUNLUK_STOK <> 0 OR GUNLUK_SATIS > 0
	) 
RETURN(@Result)
END

CREATE FUNCTION [workcube_devcatalyst_retail].[fnc_get_ortalama_satis_stok_depo]
(
@stock_id_list  NVARCHAR(MAX),
@store int,
@date1 datetime,
@date2 datetime
)
RETURNS float
AS
BEGIN
  DECLARE @Result float
  SET @Result = ( 
				SELECT TOP 1
					ROUND(workcube_devcatalyst.IS_ZERO(SUM(SATIS),0)/ workcube_devcatalyst.IS_ZERO(SUM(GECERLI_GUN),1),2) AS ORAN
				FROM
				(
					SELECT T1.*,
							CASE WHEN T1.StokMiktarı > 0 THEN  1 
								 WHEN T1.SATIS > 0 THEN 1
								 WHEN T2.StokMiktarı + T2.SATIS = T1.StokMiktarı AND (T1.StokMiktarı <= 0 ) THEN 1 else 0 end as GECERLI_GUN 
					FROM
					(
					SELECT D.FULLDATE,xxx.STOCK_ID,xxx.StokMiktarı, 
					case when D.FULLDATE = xxx.TARIH THEN xxx.SATIS ELSE 0 END AS SATIS
					FROM DimDate D CROSS  JOIN workcube_devcatalyst.fnSplit(@stock_id_list,',') AS TB1
					outer apply
					(
						select 
							TOP 1 *
						from
							STOK_GUNLUK_DURUM  t
						where
							t.STOCK_ID = TB1.item  AND t.STORE = @STORE AND D.FullDate >= t.TARIH
						order by 
							t.TARIH DESC 
					) as xxx
					WHERE
						D.FULLDATE BETWEEN @DATE1 AND @DATE2
					) AS T1

					OUTER APPLY 

					(
						select 
							TOP 1 *
						from
							STOK_GUNLUK_DURUM  t
						where
							t.STOCK_ID = T1.STOCK_ID  AND  t.STORE = @STORE AND T1.FullDate < t.TARIH AND t.SATIS > 0 
						order by 
							t.TARIH DESC 
					) T2 
				) AS LAST_TABLE
				WHERE
					STOCK_ID = @stock_id_list
				GROUP BY STOCK_ID
			)
RETURN(@Result)
END

CREATE FUNCTION [workcube_devcatalyst_retail].[fnc_get_rival_avg]
(
@product_id int
)
RETURNS float
AS
BEGIN
  DECLARE @Result float
  SET @Result =   (
          SELECT
   				AVG(P) AS PRICE
		   FROM
   				(
					SELECT 
						ISNULL(PRICE_RIVAL.PRICE,0) + ISNULL(PRICE_RIVAL.PRICE_2,0) AS P
					FROM

					WHERE
						PRICE_RIVAL.PRODUCT_ID = @product_id AND 
						PRICE_RIVAL.STARTDATE <= GETDATE() AND PRICE_RIVAL.FINISHDATE >= GETDATE()
			   ) AS RAKIP_FIYATLAR
      )

CREATE FUNCTION [workcube_devcatalyst_retail].[fnSplit](
        @sInputList VARCHAR(8000) -- List of delimited items
      , @sDelimiter VARCHAR(8000) = ',' -- delimiter that separates items
    ) RETURNS @List TABLE (item VARCHAR(8000))
    
    BEGIN
    DECLARE @sItem VARCHAR(8000)
    WHILE CHARINDEX(@sDelimiter,@sInputList,0) <> 0
     BEGIN
     SELECT
      @sItem=RTRIM(LTRIM(SUBSTRING(@sInputList,1,CHARINDEX(@sDelimiter,@sInputList,0)-1))),
      @sInputList=RTRIM(LTRIM(SUBSTRING(@sInputList,CHARINDEX(@sDelimiter,@sInputList,0)+LEN(@sDelimiter),LEN(@sInputList))))
    
     IF LEN(@sItem) > 0
      INSERT INTO @List SELECT @sItem
     END
    
    IF LEN(@sInputList) > 0
     INSERT INTO @List SELECT @sInputList -- Put the last item in
    RETURN
    END
GO

CREATE FUNCTION [workcube_devcatalyst_retail].[get_product_rate]
(
@pid int,
@date1 datetime,
@company_id int,
@project_id int,
@rate_period int
)
RETURNS nvarchar(50)
AS
BEGIN
 DECLARE @Result nvarchar(50)
  SET @Result = (
				SELECT
					CASE 
						WHEN URETICI_CAT_RATE <> '0'
							THEN URETICI_CAT_RATE
						WHEN URETICI_RATE <> '0'
							THEN URETICI_RATE
						WHEN DIRECT_RATE <> '0'
							THEN DIRECT_RATE
						WHEN URETICISIZ_CAT_RATE <> '0'
							THEN URETICISIZ_CAT_RATE
						ELSE
							'0' END AS PRODUCT_REVENUE_RATE
				FROM	
				(	
					SELECT
						ISNULL((
							SELECT TOP 1
								CAST(REVENUE_RATE AS FLOAT)
							FROM
								PROCESS_ROWS
							WHERE
								PERIOD = @rate_period AND
								PROCESS_STARTDATE <= @date1 AND
								PROCESS_FINISHDATE >= @date1 AND
								COMPANY_ID = @company_id AND
								ISNULL(PROJECT_ID,0) =  @project_id AND
								REVENUE_RATE IS NOT NULL AND
								REVENUE_RATE <> '' AND
								URETICI_ID IS NULL AND
								ROW_ID NOT IN (SELECT ROW_ID FROM PROCESS_ROWS_CATS)
						),'0') AS DIRECT_RATE,
						ISNULL((
							SELECT TOP 1
								CAST(REVENUE_RATE AS FLOAT)
							FROM
								PROCESS_ROWS
							WHERE
								PERIOD = @rate_period AND
								PROCESS_STARTDATE <= @date1 AND
								PROCESS_FINISHDATE >= @date1 AND
								COMPANY_ID = @company_id AND
								ISNULL(PROJECT_ID,0) =  @project_id AND
								URETICI_ID =  T1.SUB_TYPE_ID AND
								REVENUE_RATE IS NOT NULL AND
								REVENUE_RATE <> '' AND

CREATE FUNCTION [workcube_devcatalyst_retail].[uf_listgetat] (
    @list varchar(2000), @pos smallint, @delimiter char(1)
)
RETURNS varchar(500) 

AS
BEGIN 
    DECLARE @listitem varchar(500), @cpos smallint, @loop smallint; 
    SET @listitem = NULL; 
    SET @list = LTRIM(RTRIM(@list))+ @delimiter; 
    SET @cpos = CHARINDEX(@delimiter,@list, 1); 
    SET @loop = 0; 
    IF LEN(REPLACE(@list,@delimiter, '')) > 0 -- we have a list of some length
    BEGIN
        WHILE @cpos > 0
        BEGIN
            SET @listitem = LTRIM(RTRIM(LEFT(@list, @cpos - 1))) 
            SET @loop = @loop + 1
            IF @loop = @pos RETURN @listitem  
            SET @list = RIGHT(@list, LEN(@list) - @cpos)
            SET @cpos = CHARINDEX(@delimiter, @list, 1)
        END
    END    
    SET @listitem = NULL -- if we got here, the position specifed exceeds the list length
    RETURN @listitem 
END

CREATE PROCEDURE [workcube_devcatalyst_retail].[GET_SIP_DETAY_G]

@stock_id_list  NVARCHAR(MAX),
@DATE1 DATE,
@DATE2 DATE,
@STORE INT
AS
BEGIN
SELECT
	STOCK_ID,
	SUM(SATIS) AS SATIS,SUM(GECERLI_GUN) AS GECERLI_GUN,
	workcube_devcatalyst.IS_ZERO(SUM(SATIS),1)/ workcube_devcatalyst.IS_ZERO(SUM(GECERLI_GUN),1) AS ORAN
FROM
(
	SELECT T1.*,
			CASE WHEN T1.StokMiktarı > 0 THEN  1 
				 WHEN  T2.StokMiktarı + T2.SATIS = T1.StokMiktarı AND (T1.StokMiktarı < 0 OR T1.StokMiktarı= 0  ) THEN 1 else 0 end as GECERLI_GUN 
	FROM
	(
	SELECT D.FULLDATE,xxx.STOCK_ID,xxx.StokMiktarı, 
	case when D.FULLDATE = xxx.TARIH THEN xxx.SATIS ELSE 0 END AS SATIS
	FROM DimDate D CROSS  JOIN workcube_devcatalyst.fnSplit(@stock_id_list,',') AS TB1
	outer apply
	(
		select 
			TOP 1 *
		from
			STOK_GUNLUK_DURUM  t
		where
			D.FullDate >= t.TARIH and t.STOCK_ID = TB1.item  AND t.STORE = @STORE
		order by 
			t.TARIH DESC 
	) as xxx
	WHERE
		D.FULLDATE BETWEEN @DATE1 AND @DATE2
	) AS T1

	OUTER APPLY 

	(
		select 
			TOP 1 *
		from
			STOK_GUNLUK_DURUM  t
		where
			T1.FullDate <= t.TARIH and t.STOCK_ID = T1.STOCK_ID  AND  t.SATIS <> 0 AND t.STORE = @STORE
		order by 
			t.TARIH ASC 
	) T2 
) AS LAST_TABLE
GROUP BY STOCK_ID

END

CREATE PROCEDURE [workcube_devcatalyst_retail].[GUNLUK_STOK_AKTARIM] AS BEGIN

IF EXISTS (SELECT * FROM tempdb.sys.tables WHERE name = '##TMP2')
DROP TABLE ##TMP2

SELECT 
	SUM(STOCK_IN) AS GIREN,
	SUM(STOCK_OUT) AS CIKAN,
	SUM(STOCK_IN-STOCK_OUT) AS BAKIYE,
	CONVERT(VARCHAR(10), PROCESS_DATE, 120) AS TARIH ,
	SUM(
		CASE 
			WHEN PROCESS_TYPE IN (67,52,53,70,71,-1003,-1004,-1005) THEN STOCK_OUT 
			WHEN (UPPER_PROCESS_TYPE IS NOT NULL AND UPPER_PROCESS_TYPE IN (67,52,53,70,71,-1003,-1004,-1005)) THEN STOCK_OUT
			ELSE 0 END
		)  AS SATIS,
	STOCK_ID,
	STORE
INTO ##TMP2
FROM 
	GET_STOCK_ROWS
WHERE 
	PROCESS_DATE IS NOT NULL
	AND
	(
	(YEAR(PROCESS_DATE) > 2014 AND PROCESS_TYPE <> 114)
	OR
	YEAR(PROCESS_DATE) = 2014
	)
GROUP BY 
	CONVERT(VARCHAR(10), PROCESS_DATE, 120),
	STOCK_ID,
	STORE
	
	
	
	
IF EXISTS (SELECT * FROM sys.tables WHERE name = 'STOK_GUNLUK_DURUM')
DROP TABLE STOK_GUNLUK_DURUM

SELECT a.TARIH, a.STOCK_ID,a.STORE,a.GIREN,a.CIKAN,a.SATIS ,(SELECT SUM(b.BAKIYE)
                       FROM ##TMP2 b
                       WHERE b.STORE = a.STORE AND b.STOCK_ID = a.STOCK_ID AND  b.TARIH <= a.TARIH ) AS StokMiktarı
INTO STOK_GUNLUK_DURUM
FROM   ##TMP2 a
ORDER BY a.STOCK_ID,a.STORE,a.TARIH
	
END

	SELECT T1.*,

				 WHEN T1.SATIS > 0 THEN 1 
				 WHEN  T2.StokMiktarı + T2.SATIS = T1.StokMiktarı AND (T1.StokMiktarı <= 0  ) THEN 1 else 0 end as GECERLI_GUN 
	FROM
	(

		FROM

				where
					t.STOCK_ID = TB1.item  AND t.STORE = @STORE and D.FullDate >= t.TARIH
				order by 
					t.TARIH DESC 
			) as xxx
		WHERE
			D.FULLDATE BETWEEN @DATE1 AND @DATE2
	) AS T1

CREATE PROCEDURE [workcube_devcatalyst_retail].[pro_get_ortalama_satis_stok_new]
			(
				@sno int,
				@dept_id int,
				@date1 datetime,
				@date2 datetime
			)
		AS
		BEGIN 
					SELECT
						ROUND(SUM(TOPLAM_SATIS/TOPLAM_GUN),2)						
					FROM
						(
						SELECT
							SUM(SATIS_MIKTAR) AS TOPLAM_SATIS,
							COUNT(CASE WHEN (STOCK_MIKTAR > 0 OR SATIS_MIKTAR <> 0) THEN 1 ELSE NULL END) AS TOPLAM_GUN,
							DEPARTMENT_HEAD
						FROM
							(
								SELECT
									TARIH,
									DEPARTMENT_HEAD,
									ISNULL(SUM(SR4.STOCK_OUT - SR4.STOCK_IN),0) SATIS_MIKTAR,
									STOCK_MIKTAR
								FROM
									(
									SELECT
										T1.TARIH,
										T1.DEPARTMENT_HEAD,
										T1.DEPARTMENT_ID,
										ISNULL(SUM(SR3.STOCK_IN-SR3.STOCK_OUT),0) AS STOCK_MIKTAR
									FROM
										(
											SELECT
												PPD.TARIH,
												D.DEPARTMENT_HEAD,
												D.DEPARTMENT_ID,
												D.IS_STORE
											FROM
												PRODUCT_PRICE_DATES PPD,
												workcube_devcatalyst.DEPARTMENT D
											WHERE
												(
													   (@dept_id = 13 AND D.DEPARTMENT_ID <> 13 ) OR
													   (@dept_id <> 13 AND D.DEPARTMENT_ID = @dept_id)  
												 )
												 AND
												D.IS_STORE IN (1,3) AND
												ISNULL(D.IS_PRODUCTION,0) = 0 AND
												D.DEPARTMENT_ID NOT IN (13,17) AND
												PPD.TARIH BETWEEN @date1 AND @date2
										) T1
											LEFT JOIN GET_STOCK_ROWS SR3 ON 
												(
													SR3.STOCK_ID = @sno AND
													SR3.PROCESS_DATE <= T1.TARIH AND
													YEAR(SR3.PROCESS_DATE) = YEAR(T1.TARIH) AND
													SR3.STORE = T1.DEPARTMENT_ID
												)
									GROUP BY
										T1.TARIH,
										T1.DEPARTMENT_HEAD,
										T1.DEPARTMENT_ID
									) 
									GUNLER
										 LEFT JOIN GET_STOCK_ROWS SR4 ON 
											(
												SR4.STOCK_ID = @sno AND 
												YEAR(SR4.PROCESS_DATE) = YEAR(GUNLER.TARIH) AND
												MONTH(SR4.PROCESS_DATE) = MONTH(GUNLER.TARIH) AND
												DAY(SR4.PROCESS_DATE) = DAY(GUNLER.TARIH) AND
												SR4.PROCESS_TYPE IN (67,52,70,-1003,-1005) AND
												SR4.STORE = GUNLER.DEPARTMENT_ID
											)
								GROUP BY
									TARIH,
									DEPARTMENT_HEAD,
									STOCK_MIKTAR
							) TSON1
					   GROUP BY
							DEPARTMENT_HEAD
						 ) TSON
						WHERE
							TOPLAM_GUN > 0
      END

CREATE PROCEDURE [workcube_devcatalyst_retail].[stock_devir_hizi_getir]

@stock_id_list  NVARCHAR(MAX),
@DATE1 DATE,
@DATE2 DATE,
@STORE INT
AS
BEGIN

SELECT
	STOCK_ID,
	ROUND(CASE WHEN (BASLANGIC_STOK + BITIS_STOK) > 0 THEN SATIS_TOPLAM / ((BASLANGIC_STOK + BITIS_STOK) / 2) ELSE 0 END,2) AS STOK_DEVIR_HIZI,
	SATIS_TOPLAM,
	BASLANGIC_STOK,
	BITIS_STOK
FROM
	(
	SELECT 
		TB1.item AS STOCK_ID,
		ISNULL(SATIS_TOPLAM_TABLO.SATIS_TOPLAM,0) AS SATIS_TOPLAM,
		ISNULL(BASLANGIC_STOK_TABLO.BASLANGIC_STOK,0) AS BASLANGIC_STOK,
		ISNULL(BITIS_STOK_TABLO.BITIS_STOK,0) AS BITIS_STOK
	FROM 
		workcube_devcatalyst.fnSplit(@stock_id_list,',') AS TB1
			outer apply
			(
				select 
					SUM(SATIS) AS SATIS_TOPLAM
				from
					STOK_GUNLUK_DURUM  t
				where
					t.STOCK_ID = TB1.item  AND t.STORE = @STORE and t.TARIH >= @DATE1 AND t.TARIH <= @DATE2
			) as SATIS_TOPLAM_TABLO		
			outer apply
			(
				select 
					StokMiktarı AS BASLANGIC_STOK
				from
					STOK_GUNLUK_DURUM  t2
				where
					t2.STOCK_ID = TB1.item  AND t2.STORE = @STORE and t2.TARIH = @DATE1
			) as BASLANGIC_STOK_TABLO	
			outer apply
			(
				select 
					StokMiktarı AS BITIS_STOK
				from
					STOK_GUNLUK_DURUM  t3
				where
					t3.STOCK_ID = TB1.item  AND t3.STORE = @STORE and t3.TARIH = @DATE2
			) as BITIS_STOK_TABLO
	) T1
WHERE
	STOCK_ID IS NOT NULL
END
